package ru.itmo.tpo;

/**
 * Класс для вычисления тригонометрических функций через ряды Тейлора
 */
public class TaylorSeries {
    
    private static final int MAX_ITERATIONS = 100;
    private static final double EPSILON = 1e-10;

    /**
     * Вычисляет sin(x) через ряд Тейлора
     * sin(x) = x - x³/3! + x⁵/5! - x⁷/7! + ...
     * 
     * @param x аргумент в радианах
     * @param precision точность вычисления
     * @return значение sin(x)
     */
    public static double sin(double x, double precision) {
        // Приводим x к диапазону [-π, π]
        x = normalizeAngle(x);
        
        double sum = 0.0;
        double term = x; // Первый член ряда
        int n = 1;
        
        while (Math.abs(term) > precision && n < MAX_ITERATIONS) {
            sum += term;
            term *= -x * x / ((2 * n) * (2 * n + 1)); // Следующий член
            n++;
        }
        
        return sum;
    }

    /**
     * Вычисляет cos(x) через ряд Тейлора
     * cos(x) = 1 - x²/2! + x⁴/4! - x⁶/6! + ...
     * 
     * @param x аргумент в радианах
     * @param precision точность вычисления
     * @return значение cos(x)
     */
    public static double cos(double x, double precision) {
        // Приводим x к диапазону [-π, π]
        x = normalizeAngle(x);
        
        double sum = 0.0;
        double term = 1.0; // Первый член ряда
        int n = 1;
        
        while (Math.abs(term) > precision && n < MAX_ITERATIONS) {
            sum += term;
            term *= -x * x / ((2 * n - 1) * (2 * n)); // Следующий член
            n++;
        }
        
        return sum;
    }

    /**
     * Приводит угол к диапазону [-π, π]
     */
    private static double normalizeAngle(double x) {
        double pi = Math.PI;
        while (x > pi) {
            x -= 2 * pi;
        }
        while (x < -pi) {
            x += 2 * pi;
        }
        return x;
    }
}
