package ru.itmo.tpo;

/**
 * Класс для вычисления тангенса через степенной ряд
 */
public class Tangent {
    
    private static final double DEFAULT_PRECISION = 1e-10;

    /**
     * Вычисляет tg(x) = sin(x) / cos(x)
     * 
     * @param x аргумент в радианах
     * @param precision точность вычисления
     * @return значение tg(x)
     * @throws IllegalArgumentException если cos(x) ≈ 0 (x = π/2 + πk)
     */
    public static double tan(double x, double precision) {
        double sinValue = TaylorSeries.sin(x, precision);
        double cosValue = TaylorSeries.cos(x, precision);
        
        if (Math.abs(cosValue) < precision) {
            throw new IllegalArgumentException(
                "Tangent is undefined at x = " + x + " (cos(x) ≈ 0)"
            );
        }
        
        return sinValue / cosValue;
    }

    /**
     * Вычисляет tg(x) с точностью по умолчанию
     * 
     * @param x аргумент в радианах
     * @return значение tg(x)
     */
    public static double tan(double x) {
        return tan(x, DEFAULT_PRECISION);
    }
}
